38/*
39 * a two dimensional linked list of heur headers
40 *
41 * heur headers are heads of lists that holds all
42 * heurs with the same number of elements beneath them together
43 *
44 * heurs are doubly linked to each column header
45 *
46 * incHeur and decHeur are the main operations
47 * when a doubly under a column header is covered or uncovered, the
48 * column header's element count is appropriately incremented or
49 * decremented and the heur it's linked to is moved from one heur list
50 * to another
51 */

54 void *hcol;
55 struct heur *prev, *next;

58} Heur;

60/*
61 * columns describe the constraints
62 * if Doubly is a root, drow = d->root->rmax, dcol = d->root->cmax
63 *
64 * to reduce redundancy, if doubly is not header:
65 * 0 < drow < d->root->rmax and
66 * 0 < dcol < d->root->cmax
67 * if a header:
68 * drow >= d->root->rmax

109 * row: matrix row associated with the solTree
